Multilayer Switch (MLS)
Admin VLAN – full access
ip access-list extended ADMIN_POLICY
 remark Admin full access to all VLANs
 permit ip 192.168.10.0 0.0.0.255 any

Management VLAN – only reachable by Admin
ip access-list extended MGMT_POLICY
 remark Allow Admin to reach Management
 permit ip 192.168.10.0 0.0.0.255 192.168.20.0 0.0.0.255
 remark Deny all other sources to Management
 deny ip any 192.168.20.0 0.0.0.255
 remark Permit other outbound traffic
 permit ip any any

Users VLAN – only DHCP + ICMP replies + TCP established to Admin
ip access-list extended USER_POLICY
 remark Allow DHCP to Admin server
 permit udp 192.168.30.0 0.0.0.255 host 192.168.10.10 eq bootps
 permit udp 192.168.30.0 0.0.0.255 host 192.168.10.10 eq bootpc
 remark Allow ICMP echo-replies from Admin
 permit icmp 192.168.30.0 0.0.0.255 192.168.10.0 0.0.0.255 echo-reply
 remark Allow TCP established sessions to Admin
 permit tcp 192.168.30.0 0.0.0.255 192.168.10.0 0.0.0.255 established
 remark Deny all other traffic from Users
 deny ip 192.168.30.0 0.0.0.255 any
 remark Permit everything else (optional)
 permit ip any any

IoT VLAN – only DHCP + ICMP replies + TCP established to Admin
ip access-list extended IOT_POLICY
 remark Allow DHCP to Admin server
 permit udp 192.168.40.0 0.0.0.255 host 192.168.10.10 eq bootps
 permit udp 192.168.40.0 0.0.0.255 host 192.168.10.10 eq bootpc
 remark Allow ICMP echo-replies from Admin
 permit icmp 192.168.40.0 0.0.0.255 192.168.10.0 0.0.0.255 echo-reply
 remark Allow TCP established sessions to Admin
 permit tcp 192.168.40.0 0.0.0.255 192.168.10.0 0.0.0.255 established
 remark Deny all other traffic from IoT
 deny ip 192.168.40.0 0.0.0.255 any
 remark Permit everything else (optional)
 permit ip any any

Apply ACLs to VLAN interfaces
interface Vlan10
 description Admin VLAN (DHCP server here)
 ip address 192.168.10.1 255.255.255.0
 ip access-group ADMIN_POLICY in

interface Vlan20
 description Management VLAN
 ip address 192.168.20.1 255.255.255.0
 ip helper-address 192.168.10.10
 ip access-group MGMT_POLICY in

interface Vlan30
 description Users VLAN
 ip address 192.168.30.1 255.255.255.0
 ip helper-address 192.168.10.10
 ip access-group USER_POLICY in

interface Vlan40
 description IoT VLAN
 ip address 192.168.40.1 255.255.255.0
 ip helper-address 192.168.10.10
 ip access-group IOT_POLICY in
