Multilayer Switch (MLS)
ACL for IoT Restrictions
! Enter configuration mode
configure terminal
ip access-list extended IOT_RESTRICTIONS
  Deny traffic from IoT (192.168.40.0/24) to Admin/Servers (192.168.10.0/24)
  deny ip 192.168.40.0 0.0.0.255 192.168.10.0 0.0.0.255
  Deny traffic from IoT (192.168.40.0/24) to Management (192.168.20.0/24)
  deny ip 192.168.40.0 0.0.0.255 192.168.20.0 0.0.0.255
  permit ip any any
exit
 Apply the ACL to the IoT VLAN interface
interface Vlan40
  ip access-group IOT_RESTRICTIONS in
exit

ACL for User Restrictions
! Enter configuration mode
configure terminal
ip access-list extended USER_TO_ADMIN_POLICY
  ! Permit traffic from the User network (192.168.30.0/24) to the specific jump-host
  permit ip 192.168.30.0 0.0.0.255 host 192.168.10.10
  ! Deny all other traffic from the User network to the entire Admin/Servers network
  deny ip 192.168.30.0 0.0.0.255 192.168.10.0 0.0.0.255
  permit ip any any
exit
 Apply the ACL to the User VLAN interface
interface Vlan30
  ip access-group USER_TO_ADMIN_POLICY in
exit

Deny Outside Access
Router0
! Enter configuration mode
configure terminal
ip access-list extended OUTSIDE_IN_BLOCK
  ! Deny any traffic from the outside attempting to reach your internal subnets
  deny ip any 192.168.10.0 0.0.0.255
  deny ip any 192.168.20.0 0.0.0.255
  deny ip any 192.168.30.0 0.0.0.255
  deny ip any 192.168.40.0 0.0.0.255
  permit ip any any
exit
 Apply the ACL to the outside-facing interface
Int fa0/0
  ip access-group OUTSIDE_IN_BLOCK in
exit
