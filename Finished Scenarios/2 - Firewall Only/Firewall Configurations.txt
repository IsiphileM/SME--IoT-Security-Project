 nameif transit
 security-level 50
 ip address 10.0.0.2 255.255.255.252
 no shutdown

route outside 0.0.0.0 0.0.0.0 203.0.113.1

! NAT: allow inside (LAN) to reach Internet via ASA outside address
object network OBJ_LAN
 subnet 192.168.1.0 255.255.255.0
 nat (transit,outside) dynamic interface

! Static NAT (port-forward) - expose internal web server (optional)
object network OBJ_WEB
 host 192.168.1.100
 nat (transit,outside) static interface service tcp 80 80

access-list OUTSIDE_IN extended permit icmp any host 203.0.113.10 echo
access-list OUTSIDE_IN extended deny ip any any
